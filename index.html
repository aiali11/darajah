import cv2 import numpy as np import json from js import document, window, Uint8ClampedArray, ImageData, alert, console # عند اكتمال تحميل بايثون document.getElementById('loadingOverlay').style.display = 'none' document.getElementById('scanKeyBtn').disabled = False document.getElementById('setupKeyBtn').disabled = False def process_image(*args): try: document.getElementById('loadingText').innerText = "Python يعالج الصورة..." # 1. سحب الصورة من الكانفاس (الجافاسكريبت) إلى البايثون canvas = document.getElementById('canvasOutput') ctx = canvas.getContext('2d') w, h = canvas.width, canvas.height img_data = ctx.getImageData(0, 0, w, h) # تحويل البيانات إلى مصفوفة Numpy data = img_data.data.to_py() np_arr = np.array(data, dtype=np.uint8).reshape((h, w, 4)) gray = cv2.cvtColor(np_arr, cv2.COLOR_RGBA2GRAY) # 2. فلترة قوية واكتشاف الحواف blurred = cv2.GaussianBlur(gray, (5, 5), 0) thresh = cv2.adaptiveThreshold(blurred, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY_INV, 51, 15) kernel = np.ones((3,3), np.uint8) dilated = cv2.dilate(thresh, kernel, iterations=1) contours, _ = cv2.findContours(dilated, cv2.RETR_LIST, cv2.CHAIN_APPROX_SIMPLE) # 3. صيد المربعات الأربعة بذكاء البايثون (يتجاهل الكيبورد والمفرش) square_contours = [] img_area = w * h for cnt in contours: area = cv2.contourArea(cnt) if (img_area * 0.001) < area < (img_area * 0.05): peri = cv2.arcLength(cnt, True) approx = cv2.approxPolyDP(cnt, 0.04 * peri, True) if 4 <= len(approx) <= 6: x, y, bw, bh = cv2.boundingRect(approx) aspect = float(bw) / bh if 0.5 <= aspect <= 1.5: M = cv2.moments(cnt) if M["m00"] != 0: cx = int(M["m10"] / M["m00"]) cy = int(M["m01"] / M["m00"]) square_contours.append({"cx": cx, "cy": cy, "area": area}) if len(square_contours) < 4: raise Exception("لم يتم العثور على المربعات الأربعة.") # ترتيب المربعات حسب المساحة لأخذ أكبر 10 فقط وتجاهل الغبار والأزرار square_contours.sort(key=lambda x: x["area"], reverse=True) top_candidates = square_contours[:10] # استخراج الزوايا tl = min(top_candidates, key=lambda p: p["cx"] + p["cy"]) br = max(top_candidates, key=lambda p: p["cx"] + p["cy"]) tr = max(top_candidates, key=lambda p: p["cx"] - p["cy"]) bl = min(top_candidates, key=lambda p: p["cx"] - p["cy"]) # 4. فرد الورقة رياضياً (Perspective Transform) pts1 = np.float32([[tl["cx"], tl["cy"]], [tr["cx"], tr["cy"]], [br["cx"], br["cy"]], [bl["cx"], bl["cy"]]]) maxWidth = max(int(np.linalg.norm(np.array([br["cx"], br["cy"]]) - np.array([bl["cx"], bl["cy"]]))), int(np.linalg.norm(np.array([tr["cx"], tr["cy"]]) - np.array([tl["cx"], tl["cy"]])))) maxHeight = max(int(np.linalg.norm(np.array([tr["cx"], tr["cy"]]) - np.array([br["cx"], br["cy"]]))), int(np.linalg.norm(np.array([tl["cx"], tl["cy"]]) - np.array([bl["cx"], bl["cy"]])))) pts2 = np.float32([[0, 0], [maxWidth, 0], [maxWidth, maxHeight], [0, maxHeight]]) M_trans = cv2.getPerspectiveTransform(pts1, pts2) warped = cv2.warpPerspective(gray, M_trans, (maxWidth, maxHeight)) # 5. قراءة التظليل من الدوائر المفرغة البيضاء thresh_ans = cv2.adaptiveThreshold(warped, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY_INV, 41, 10) proof_img = cv2.cvtColor(warped, cv2.COLOR_GRAY2RGBA) qCount = int(document.getElementById('numQuestions').value) cCount = int(document.getElementById('numChoices').value) gridX = int(maxWidth * 0.04) gridY = int(maxHeight * 0.07) gridW = int(maxWidth * 0.92) gridH = int(maxHeight * 0.89) qGridY = int(gridY + (gridH * 0.07)) qGridH = int(gridH * 0.93) rowsPerCol = int(np.ceil(qCount / 2.0)) colW = int(gridW / 2) rowH = int(qGridH / rowsPerCol) detected_answers = [] for q in range(qCount): isRightCol = (q % 2 == 0) rowIdx = q // 2 colRightEdge = gridX + (gridW if isRightCol else colW) rowCenterY = int(qGridY + (rowIdx * rowH) + (rowH / 2)) selected_indices = [] for c in range(cCount): percentFromRight = 0.20 + (c * (0.80 / cCount)) + (0.80 / cCount / 2) centerX = int(colRightEdge - (colW * percentFromRight)) visualRoiSize = min(colW * 0.12, rowH * 0.8) vx = int(centerX - (visualRoiSize / 2)) vy = int(rowCenterY - (visualRoiSize / 2)) # البؤرة المركزية (القلب الأبيض) coreSize = visualRoiSize * 0.40 x = int(centerX - (coreSize / 2)) y = int(rowCenterY - (coreSize / 2)) w_roi = int(coreSize) h_roi = int(coreSize) if x > 0 and y > 0 and x + w_roi < thresh_ans.shape[1] and y + h_roi < thresh_ans.shape[0]: roi = thresh_ans[y:y+h_roi, x:x+w_roi] black_pixels = cv2.countNonZero(roi) total_pixels = w_roi * h_roi fill_percentage = black_pixels / total_pixels # إذا كان الحبر يغطي أكثر من 10% من قلب الدائرة البيضاء فهو تظليل مؤكد if fill_percentage > 0.10: selected_indices.append(c) cv2.rectangle(proof_img, (vx, vy), (vx + int(visualRoiSize), vy + int(visualRoiSize)), (0, 255, 0, 255), 3) else: cv2.rectangle(proof_img, (vx, vy), (vx + int(visualRoiSize), vy + int(visualRoiSize)), (255, 0, 0, 100), 1) if len(selected_indices) == 1: detected_answers.append(selected_indices[0]) elif len(selected_indices) > 1: detected_answers.append(-2) else: detected_answers.append(-1) # 6. إرجاع الصورة المصححة إلى الجافاسكريبت لعرضها proof_flat = proof_img.flatten().tolist() js_array = Uint8ClampedArray.new(proof_flat) new_img_data = ImageData.new(js_array, maxWidth, maxHeight) proofCanvas = document.getElementById('proofCanvas') proofCanvas.width = maxWidth proofCanvas.height = maxHeight proofCanvas.getContext('2d').putImageData(new_img_data, 0, 0) document.getElementById('loadingOverlay').style.display = 'none' # استدعاء دالة عرض النتائج بالجافاسكريبت window.finishPythonProcess(json.dumps(detected_answers)) except Exception as e: console.log(str(e)) window.handlePythonError()
